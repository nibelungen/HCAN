00039 void canix_SFP_HMS_handler(const canix_frame *frame)
00040 {
00041     canix_frame answer;
00042 
00043     answer.src = canix_selfaddr();
00044     answer.dst = frame->src;
00045     answer.proto = HCAN_PROTO_SFP;
00046     answer.data[0] = HCAN_SRV_HMS;
00047 
00048 
00049     switch (frame->data[1])
00050     {
00051         case HCAN_HMS_PING_REQUEST :
00052             answer.data[1] = HCAN_HMS_PING_REPLAY;
00053             answer.size = 2;
00054             canix_frame_send(&answer);
00055             break;
00056         case HCAN_HMS_DEVICE_STATE_REQUEST :
00057             answer.data[1] = HCAN_HMS_DEVICE_STATE_REPLAY;
00058             answer.data[2] = HCAN_HMS_DEV_STATE_APPLICATION;
00059             answer.size = 3;
00060             canix_frame_send(&answer);
00061             break;
00062         case HCAN_HMS_DEVICE_RESET :
00063             canix_reset();
00064             break;
00065         case HCAN_HMS_DEVICE_TYPE_REQUEST :
00066             answer.data[1] = HCAN_HMS_DEVICE_TYPE_REPLAY;
00067 #if defined (__AVR_ATmega8__)
00068             answer.data[2] = 0;
00069 #elif defined (__AVR_ATmega32__)
00070             answer.data[2] = 1;
00071 #endif
00072             answer.data[3] = eeprom_read_byte((uint8_t *)EEPR_BOARD_TYPE);
00073             answer.size = 4;
00074             canix_frame_send(&answer);
00075             break;
00076         case HCAN_HMS_INTERNAL_EEPROM_WRITE :
00077             {
00078                 // only write to eeprom, if the value changed !
00079                 uint8_t read = 
00080                     eeprom_read_byte((uint8_t *)(frame->data[2] |
00081                                 (frame->data[3] << 8)));
00082 
00083                 if (read != frame->data[4])
00084                     eeprom_write_byte((uint8_t *)(frame->data[2] |
00085                                 (frame->data[3] << 8)), frame->data[4]);
00086 
00087                 answer.data[1] = HCAN_HMS_INTERNAL_EEPROM_WRITE_ACK;
00088                 answer.size = 2;
00089                 canix_frame_send(&answer);
00090             }
00091             break;
00092         case HCAN_HMS_INTERNAL_EEPROM_READ_REQUEST :
00093             answer.data[1] = HCAN_HMS_INTERNAL_EEPROM_READ_REPLAY;
00094             answer.data[2] = eeprom_read_byte((uint8_t *)(frame->data[2] |
00095                         (frame->data[3] << 8)));
00096             answer.size = 3;
00097             canix_frame_send(&answer);
00098             break;
00099         case HCAN_HMS_INTERNAL_EEPROM_READ_BULK_REQUEST :
00100             {
00101                 uint16_t address;
00102                 address = frame->data[2] | (frame->data[3] << 8);
00103                 answer.data[1] = HCAN_HMS_INTERNAL_EEPROM_READ_BULK_REPLAY;
00104                 answer.data[2] = eeprom_read_byte((uint8_t*) (address + 0));
00105                 answer.data[3] = eeprom_read_byte((uint8_t*) (address + 1));
00106                 answer.data[4] = eeprom_read_byte((uint8_t*) (address + 2));
00107                 answer.data[5] = eeprom_read_byte((uint8_t*) (address + 3));
00108                 answer.data[6] = eeprom_read_byte((uint8_t*) (address + 4));
00109                 answer.data[7] = eeprom_read_byte((uint8_t*) (address + 5));
00110                 answer.size = 8;
00111                 canix_frame_send(&answer);
00112             }
00113             break;
00114         case HCAN_HMS_UPTIME_QUERY :
00115             answer.size = 6;
00116             answer.data[1] = HCAN_HMS_UPTIME_REPLAY;
00117             answer.data[2] = canix_rtc_clock.uptime_sec >> 24;
00118             answer.data[3] = canix_rtc_clock.uptime_sec >> 16;
00119             answer.data[4] = canix_rtc_clock.uptime_sec >> 8;
00120             answer.data[5] = canix_rtc_clock.uptime_sec;
00121             canix_frame_send(&answer);
00122             break;
00123         case HCAN_HMS_SYSLOG_LEVEL_SET :
00124 #ifdef CANIX_SYSLOG
00125             canix_set_syslog_level(frame->data[2]);
00126             eeprom_write_byte((uint8_t *)EEPR_DEBUG_LEVEL, frame->data[2]);
00127 #endif
00128             break;
00129         case HCAN_HMS_BUILD_VERSION_QUERY :
00130             answer.data[1] = HCAN_HMS_BUILD_VERSION_REPLAY;
00131             answer.data[2] = 
00132 #include <buildver> 
00133                 >> 8;
00134             answer.data[3] = 
00135 #include <buildver>
00136                 ;
00137             answer.size = 4;
00138             canix_frame_send(&answer);
00139             break;
00140         case HCAN_HMS_CAN_EC_QUERY :
00141             {
00142                 uint16_t ec = can_error_counters();
00143                 answer.data[1] = HCAN_HMS_CAN_EC_REPLAY;
00144                 answer.data[2] = ec >> 8;
00145                 answer.data[3] = ec;
00146                 answer.size = 4;

00148             }
00149             break;

00184             {
00185                 answer.data[1] = HCAN_HMS_TX_DROPPED_REPLAY;

00187                 answer.data[3] = canix_tx_dropped >> 16;
00188                 answer.data[4] = canix_tx_dropped >> 8;
