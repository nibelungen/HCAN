00420 void canix_deliver_frame(const canix_frame *frame)
00421 {
00422     uint8_t i;
00423 
00424     // nachschauen, ob dafuer ein callback registiert ist:
00425     for (i = 0; i < MAX_FRAME_CALLBACKS; i++)
00426     {
00427         if (frame_callback_entry[i].callback)
00428         {
00429             // dieser Callback ist installiert; nun schauen,
00430             // ob er passt:
00431             if ((frame_callback_entry[i].dst == -1) ||
00432                     ((frame_callback_entry[i].dst != -1) &&
00433                      (frame_callback_entry[i].dst == frame->dst)))
00434             {
00435                 // Adressfilter haben gematched oder stehen auf ANY;
00436                 
00437                 // Schauen, ob das Protokoll und der
00438                 // Service passen:
00439                 if (
00440                         ((frame_callback_entry[i].proto == 0) || 
00441                         (frame_callback_entry[i].proto == frame->proto)) &&
00442                         ((frame_callback_entry[i].srv == 0) ||
00443                         (frame_callback_entry[i].srv == frame->data[0])))
00444                 {
00445                     frame_callback_entry[i].callback(frame);
00446                 }
00447             }
00448         }

00524 {
00525     uint8_t i;
00526 
00527     for (i = 0; i < MAX_IDLE_CALLBACKS; i++)
00528     {
00529         if (can_message_available())
00530         {
00531             uint32_t extid;
00532             canix_frame frame;
00533 
00534             // Frame einlesen

00536             frame.dst = (extid >> 10) & ADDR_MASK;
00537             frame.src = extid & ADDR_MASK;

00543                 idle_callback[i]();
