00345     assert(m_page_size);
00346 
00347     // Das zu flashende Hexfile laden
00348     ihexfile hexfile;
00349     hexfile.load(filename,m_page_size);

00351

00356 
00357     bool incremental_flash_active = false;
00358 
00359     ihexfile cached_hexfile;
00360     if (incremental_flash)
00361     {
00362         try
00363         {
00364             cmd_reset();
00365             sleep(1);
00366 
00367             m_tcon.send_DEVICE_BOOT_STOP(m_bcon.src(), m_bcon.dst());
00368             m_tcon.recv_DEVICE_BOOT_STOP_ACK(m_bcon.dst(), m_bcon.src());

00370             assert(cmd_getstate() == 1);
00371 
00372             // gecachete Firmware laden
00373             cached_hexfile.load(cached_filename(), m_page_size);
00374 
00375             // Pruefsumme der Firmware des Target-Boards besorgen...
00376             uint16_t board_crc = load_crc16_from_board(size);
00377 
00378             // Pruefsumme mit der des gecachten Files vergleichen;
00379             // stimmen sie, so kann ein inkrementelles Flashen stattfinden:
00380             if (board_crc == cached_hexfile.crc16())
00381             {
00382                 incremental_flash_active = true;
00383                 cout << "cached file crc16 is correct, incremental mode enabled" << endl;
00384             }
00385             else
00386             {
00387                 incremental_flash_active = false;
00388                 cout << "cached file crc16 is not correct, incremental mode disabled" << endl;
00389             }
00390         }
00391         catch (ihexfile_error &e)
00392         {
00393             cout << "no cache file found. Using normal flash method..." << endl;
00394         }
00395     }
00396 
00397     
00398     // 3 Versuche; falls ein Timeout auftritt, wird nochmals versucht...
00399     int n_trys = 3;
00400 
00401     while (n_trys--)
00402     {
00403         try
00404         {
00405             /*
00406              * Flash-Strategie:
00407              *
00408              * Der gesamte zu flashende Bereich (size) wird in
00409              * Bloecke aufgeteilt. Jeder Block ist eine Groesse von m_page_size
00410              * und ist vom Typ der MCU abhaengig (z.Z. nur Atmega32).
00411              *
00412              * Jeder Block muss auf der MCU "auf einen Rutsch" ins Flash
00413              * geschrieben werden. Dazu wird der Block bei der Uebertragung
00414              * in sogenannte Lines zu je 16 Bytes aufgeteilt und uebertragen.
00415              * Jede Line wird mit 4 FLASH_BUFFER_FILL Frames uebertragen und
00416              * mit einem FLASH_BUFFER_FILL_ACK vom Board bestaetigt.
00417              *
00418              * Ist der komplette Block, bestehend aus mehreren Lines, uebertragen,
00419              * so wird er auf der MCU mit FLASH_BUFFER_WRITE ins Flash
00420              * geschrieben. Die MCU bestaetigt mit FLASH_BUFFER_WRITE_ACK.
00421              *
00422              * Inkrementeller Modus:
00423              *
00424              * Hier wird nun jeder Block vor der Uebertragung mit der
00425              * gecachten Version verglichen; unterscheidet sich auch nur ein
00426              * Byte, so wird er komplett uebertragen.
00427              *
00428              */
00429             for (uint32_t block = 0; block < (size / m_page_size); block++)
00430             {
00431                 int offset = block * m_page_size;
00432 
00433                 if (incremental_flash_active)
00434                 {
00435                     // Block mit der gecachten Version vergleichen:
00436 
00437                     bool blocks_equal = true;
00438                     for (int i = 0; i < m_page_size; i++)
00439                     {
00440                         if (data[offset+i] != cached_hexfile.data()[offset+i])
00441                         {
00442                             blocks_equal = false;
00443                             break;
00444                         }
00445                     }
00446 
00447                     if (blocks_equal)
00448                     {
00449                         cout << "O";
00450                         cout.flush();
00451 
00452                         // diesen Block auslassen:
00453                         continue;
00454                     }
00455 
00456                 }
00457 
00458                 uint8_t index = 0;
00459 
00460                 // send X * 16 = m_page_size bytes en block 
00461                 for (int line = 0; line < (m_page_size / 16); line++)
00462                 {
00463                     // send 4 buffers a 4 bytes (one line) = 16 bytes
00464                     for (uint32_t i = 0; i < 4; i++)
00465                     {
00466                         m_tcon.send_FLASH_BUFFER_FILL(
00467                                 m_bcon.src(), m_bcon.dst(), index, 
00468                                 data[offset + index],
00469                                 data[offset + index + 1],
00470                                 data[offset + index + 2],
00471                                 data[offset + index + 3]);
00472 
00473                         index += 4;
00474                     }
00475 
00476                     // afer each complete line, wait for ACK
00477                     m_tcon.recv_FLASH_BUFFER_FILL_ACK(m_bcon.dst(), 
00478                             m_bcon.src());

00482                 const uint8_t hi = (block * m_page_size) >> 8;

00485                         lo, hi);
00486 
00487                 // wait for write ACK
00488 
00489                 m_tcon.recv_FLASH_BUFFER_WRITE_ACK(m_bcon.dst(), m_bcon.src());
00490 
00491                 cout << "*"; 
00492                 cout.flush();
00493             }
00494 
00495             cout << endl;
00496 
00497             // Nun das Board die CRC16 Checksumme berechnen lassen und
00498             // vergleichen
00499             try
00500             {
00501                 uint16_t board_crc = load_crc16_from_board(size);
00502                 save_crc16_to_board(board_crc,size);
00503                 cout << "written data crc16 is " << board_crc << endl;
00504 
00505                 if (board_crc != hexfile.crc16())
00506                 {
00507                     throw traceable_error(
00508                             "Critical error: crc16 NOT identical: update bootloader or maybe flash error?");
00509                 }
00510 
00511             }
00512             catch (transport_error &e)
00513             {
00514                 cerr << endl;
00515                 cerr << "warning: no firmware crc16 check support "
00516                     "(update bootloader!)" << endl;
00517             }
00518 
00519             // Firmware laden
00520             m_tcon.send_DEVICE_LOAD_APP(m_bcon.src(), m_bcon.dst());
00521 
00522 
00523             // Hexfile in den Cache kopieren, damit das inkrementelle Flashen
00524             // beim naechsen Mal moeglich ist:
00525             hexfile.copy_original_to(cached_filename());
00526 
00527             cout << endl;
00528             return;
00529         }
00530         catch (transport_error &e)
00531         {
